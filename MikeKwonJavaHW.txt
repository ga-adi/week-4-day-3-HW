public String delDel(String str) {
  String result = str;
  
  if(result.length()>3){
  if(result.charAt(1)=='d'
  && result.charAt(2)=='e'
  && result.charAt(3)=='l') {
  result = result.charAt(0) + result.substring(4);
  }}
  
  return result;
}
